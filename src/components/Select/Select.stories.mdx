import { Canvas, Meta, Story } from "@storybook/addon-docs";
import Select from "./Select";

<Meta title="Select" component={ManagedSelect} />

export const ManagedSelect = ({ initialValue = "newest", focus }) => {
  const [value, setValue] = React.useState(initialValue);
  return (
    <>
      <label
        htmlFor="filter-by"
        style={{ display: "block", marginBottom: "0.5em" }}
      >
        Filter by:
      </label>
      <Select
        id="filter-by"
        label="Sort"
        value={value}
        onChange={(ev) => setValue(ev.target.value)}
      >
        <option value="newest">Newest Releases</option>
        <option value="long">Really long selected option!</option>
        <option value="price">Price</option>
        <option value="curated">Curated</option>
      </Select>
    </>
  );
};

export const Template = (args) => <ManagedSelect {...args} />;
